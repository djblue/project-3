int fib (int n) {
    int ans[100];
    for (int i = 0; i < n; i++) {
        if (i == 0 || i = 1) {
            ans[i] = 1;
        } else {
            ans[i] = ans[i-1] + ans[i-2];
        }
    }

    return ans;
}

int main (int argc, char* argv[]) {

    int input;
    printf("Please enter a number >> ");
    scanf("%d", &input);

    printf("\"Fibonacci(%d)\" = %d\n", input, fib(input));

    return 0;
}
